15inline bool has_hex_prefix_unsafe(std::string_view input) {
16 // This is actually efficient code, see has_hex_prefix for the assembly.
17 uint32_t value_one = 1;
18 bool is_little_endian = (reinterpret_cast<char*>(&value_one)[0] == 1);

20 std::memcpy(&word0x, "0x", 2); // we would use bit_cast in C++20 and the

23 std::memcpy(&two_first_bytes, input.data(), 2);
