64ada_really_inline constexpr uint8_t path_signature(
65 std::string_view input) noexcept {
66 // The path percent-encode set is the query percent-encode set and U+003F (?),
67 // U+0060 (`), U+007B ({), and U+007D (}). The query percent-encode set is the
68 // C0 control percent-encode set and U+0020 SPACE, U+0022 ("), U+0023 (#),
69 // U+003C (<), and U+003E (>). The C0 control percent-encode set are the C0
70 // controls and all code points greater than U+007E (~).
71 size_t i = 0;
72 uint8_t accumulator{};
73 for (; i + 7 < input.size(); i += 8) {
74 accumulator |= uint8_t(path_signature_table[uint8_t(input[i])] |
75 path_signature_table[uint8_t(input[i + 1])] |
76 path_signature_table[uint8_t(input[i + 2])] |
77 path_signature_table[uint8_t(input[i + 3])] |
78 path_signature_table[uint8_t(input[i + 4])] |
79 path_signature_table[uint8_t(input[i + 5])] |
80 path_signature_table[uint8_t(input[i + 6])] |
81 path_signature_table[uint8_t(input[i + 7])]);
82 }
83 for (; i < input.size(); i++) {
84 accumulator |= uint8_t(path_signature_table[uint8_t(input[i])]);
85 }
86 return accumulator;
87}

89ada_really_inline constexpr bool verify_dns_length(
