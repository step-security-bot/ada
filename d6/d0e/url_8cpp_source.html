25bool url::parse_ipv4(std::string_view input) {
26 ada_log("parse_ipv4 ", input, "[", input.size(), " bytes]");
27 if (input.back() == '.') {
28 input.remove_suffix(1);
29 }
30 size_t digit_count{0};
31 int pure_decimal_count = 0; // entries that are decimal
32 std::string_view original_input =
33 input; // we might use this if pure_decimal_count == 4.
34 uint64_t ipv4{0};
35 // we could unroll for better performance?
36 for (; (digit_count < 4) && !(input.empty()); digit_count++) {
37 uint32_t
38 segment_result{}; // If any number exceeds 32 bits, we have an error.
39 bool is_hex = checkers::has_hex_prefix(input);
40 if (is_hex && ((input.length() == 2) ||
41 ((input.length() > 2) && (input[2] == '.')))) {
42 // special case
43 segment_result = 0;
44 input.remove_prefix(2);
45 } else {
46 std::from_chars_result r;
47 if (is_hex) {
48 r = std::from_chars(input.data() + 2, input.data() + input.size(),
49 segment_result, 16);
50 } else if ((input.length() >= 2) && input[0] == '0' &&
51 checkers::is_digit(input[1])) {
52 r = std::from_chars(input.data() + 1, input.data() + input.size(),
53 segment_result, 8);
54 } else {
55 pure_decimal_count++;
56 r = std::from_chars(input.data(), input.data() + input.size(),
57 segment_result, 10);
58 }
59 if (r.ec != std::errc()) {
60 return is_valid = false;
61 }
62 input.remove_prefix(r.ptr - input.data());
63 }
64 if (input.empty()) {
65 // We have the last value.
66 // At this stage, ipv4 contains digit_count*8 bits.
67 // So we have 32-digit_count*8 bits left.
68 if (segment_result > (uint64_t(1) << (32 - digit_count * 8))) {
69 return is_valid = false;
70 }
71 ipv4 <<= (32 - digit_count * 8);
72 ipv4 |= segment_result;
73 goto final;
74 } else {
75 // There is more, so that the value must no be larger than 255
76 // and we must have a '.'.
77 if ((segment_result > 255) || (input[0] != '.')) {
78 return is_valid = false;
79 }
80 ipv4 <<= 8;
81 ipv4 |= segment_result;
82 input.remove_prefix(1); // remove '.'
83 }
84 }
85 if ((digit_count != 4) || (!input.empty())) {
86 return is_valid = false;
87 }
88final:
89 // We could also check r.ptr to see where the parsing ended.
90 if (pure_decimal_count == 4) {
91 host = original_input; // The original input was already all decimal and we
92 // validated it.
93 } else {
94 host = ada::serializers::ipv4(ipv4); // We have to reserialize the address.
95 }
96 return true;
97}

99bool url::parse_ipv6(std::string_view input) {
100 ada_log("parse_ipv6 ", input, "[", input.size(), " bytes]");
101

103 return is_valid = false;
104 }
105 // Let address be a new IPv6 address whose IPv6 pieces are all 0.
106 std::array<uint16_t, 8> address{};
107
108 // Let pieceIndex be 0.
109 int piece_index = 0;
110
111 // Let compress be null.
112 std::optional<int> compress{};
113
114 // Let pointer be a pointer for input.
115 std::string_view::iterator pointer = input.begin();
116
117 // If c is U+003A (:), then:
118 if (input[0] == ':') {
119 // If remaining does not start with U+003A (:), validation error, return
120 // failure.
121 if (input.size() == 1 || input[1] != ':') {
122 ada_log("parse_ipv6 starts with : but the rest does not start with :");
123 return is_valid = false;
124 }
125
126 // Increase pointer by 2.
127 pointer += 2;
128
129 // Increase pieceIndex by 1 and then set compress to pieceIndex.
130 compress = ++piece_index;
131 }
132
133 // While c is not the EOF code point:
134 while (pointer != input.end()) {
135 // If pieceIndex is 8, validation error, return failure.
136 if (piece_index == 8) {
137 ada_log("parse_ipv6 piece_index == 8");
138 return is_valid = false;
139 }
140
141 // If c is U+003A (:), then:
142 if (*pointer == ':') {
143 // If compress is non-null, validation error, return failure.
144 if (compress.has_value()) {
145 ada_log("parse_ipv6 compress is non-null");
146 return is_valid = false;
147 }
148
149 // Increase pointer and pieceIndex by 1, set compress to pieceIndex, and
150 // then continue.
151 pointer++;
152 compress = ++piece_index;
153 continue;
154 }
155
156 // Let value and length be 0.
157 uint16_t value = 0, length = 0;
158
159 // While length is less than 4 and c is an ASCII hex digit,
160 // set value to value × 0x10 + c interpreted as hexadecimal number, and
161 // increase pointer and length by 1.
162 while (length < 4 && pointer != input.end() &&
163 unicode::is_ascii_hex_digit(*pointer)) {
164 // https://stackoverflow.com/questions/39060852/why-does-the-addition-of-two-shorts-return-an-int
165 value = uint16_t(value * 0x10 + unicode::convert_hex_to_binary(*pointer));
166 pointer++;
167 length++;
168 }
169
170 // If c is U+002E (.), then:
171 if (pointer != input.end() && *pointer == '.') {
172 // If length is 0, validation error, return failure.
173 if (length == 0) {
174 ada_log("parse_ipv6 length is 0");
175 return is_valid = false;
176 }
177
178 // Decrease pointer by length.
179 pointer -= length;
180
181 // If pieceIndex is greater than 6, validation error, return failure.
182 if (piece_index > 6) {
183 ada_log("parse_ipv6 piece_index > 6");
184 return is_valid = false;
185 }
186
187 // Let numbersSeen be 0.
188 int numbers_seen = 0;

190 // While c is not the EOF code point:
191 while (pointer != input.end()) {
192 // Let ipv4Piece be null.
193 std::optional<uint16_t> ipv4_piece{};
194
195 // If numbersSeen is greater than 0, then:
196 if (numbers_seen > 0) {
197 // If c is a U+002E (.) and numbersSeen is less than 4, then increase
198 // pointer by 1.
199 if (*pointer == '.' && numbers_seen < 4) {
200 pointer++;
201 }
202 // Otherwise, validation error, return failure.
203 else {

206 }
207 }
208

210 if (pointer == input.end() || !checkers::is_digit(*pointer)) {
211 ada_log(
212 "parse_ipv6 If c is not an ASCII digit, validation error, return "
213 "failure");
214 return is_valid = false;
215 }
216
217 // While c is an ASCII digit:
218 while (pointer != input.end() && checkers::is_digit(*pointer)) {
219 // Let number be c interpreted as decimal number.
220 int number = *pointer - '0';
221
222 // If ipv4Piece is null, then set ipv4Piece to number.
223 if (!ipv4_piece.has_value()) {
224 ipv4_piece = number;
225 }
226 // Otherwise, if ipv4Piece is 0, validation error, return failure.
227 else if (ipv4_piece == 0) {
228 ada_log("parse_ipv6 if ipv4Piece is 0, validation error");
229 return is_valid = false;
230 }
231 // Otherwise, set ipv4Piece to ipv4Piece × 10 + number.
232 else {
233 ipv4_piece = *ipv4_piece * 10 + number;
234 }
235
236 // If ipv4Piece is greater than 255, validation error, return failure.
237 if (ipv4_piece > 255) {
238 ada_log("parse_ipv6 ipv4_piece > 255");
239 return is_valid = false;
240 }
241
242 // Increase pointer by 1.
243 pointer++;
244 }
245
246 // Set address[pieceIndex] to address[pieceIndex] × 0x100 + ipv4Piece.
247 // https://stackoverflow.com/questions/39060852/why-does-the-addition-of-two-shorts-return-an-int
248 address[piece_index] =
249 uint16_t(address[piece_index] * 0x100 + *ipv4_piece);
250
251 // Increase numbersSeen by 1.
252 numbers_seen++;
253
254 // If numbersSeen is 2 or 4, then increase pieceIndex by 1.
255 if (numbers_seen == 2 || numbers_seen == 4) {
256 piece_index++;
257 }
258 }
259
260 // If numbersSeen is not 4, validation error, return failure.
261 if (numbers_seen != 4) {
262 return is_valid = false;
263 }
264
265 // Break.
266 break;
267 }
268 // Otherwise, if c is U+003A (:):
269 else if ((pointer != input.end()) && (*pointer == ':')) {
270 // Increase pointer by 1.
271 pointer++;
272
273 // If c is the EOF code point, validation error, return failure.
274 if (pointer == input.end()) {
275 ada_log(
276 "parse_ipv6 If c is the EOF code point, validation error, return "
277 "failure");
278 return is_valid = false;
279 }
280 }
281 // Otherwise, if c is not the EOF code point, validation error, return
282 // failure.
283 else if (pointer != input.end()) {
284 ada_log(
285 "parse_ipv6 Otherwise, if c is not the EOF code point, validation "
286 "error, return failure");
287 return is_valid = false;
288 }
289
290 // Set address[pieceIndex] to value.
291 address[piece_index] = value;
292
293 // Increase pieceIndex by 1.
294 piece_index++;
295 }
296
297 // If compress is non-null, then:
298 if (compress.has_value()) {
299 // Let swaps be pieceIndex − compress.
300 int swaps = piece_index - *compress;
301
302 // Set pieceIndex to 7.
303 piece_index = 7;
304
305 // While pieceIndex is not 0 and swaps is greater than 0,
306 // swap address[pieceIndex] with address[compress + swaps − 1], and then
307 // decrease both pieceIndex and swaps by 1.
308 while (piece_index != 0 && swaps > 0) {
309 std::swap(address[piece_index], address[*compress + swaps - 1]);
310 piece_index--;
311 swaps--;
312 }
313 }
314 // Otherwise, if compress is null and pieceIndex is not 8, validation error,
315 // return failure.
316 else if (piece_index != 8) {
317 ada_log(
318 "parse_ipv6 if compress is null and pieceIndex is not 8, validation "
319 "error, return failure");
320 return is_valid = false;
321 }
322 host = ada::serializers::ipv6(address);
323 ada_log("parse_ipv6 ", *host);
324 return true;

421ada_really_inline bool url::parse_host(std::string_view input) {
422 ada_log("parse_host ", input, "[", input.size(), " bytes]");
423 if (input.empty()) {
424 return is_valid = false;
425 } // technically unnecessary.
426 // If input starts with U+005B ([), then:
427 if (input[0] == '[') {
428 // If input does not end with U+005D (]), validation error, return failure.
429 if (input.back() != ']') {
430 return is_valid = false;
431 }
432 ada_log("parse_host ipv6");
433
434 // Return the result of IPv6 parsing input with its leading U+005B ([) and
435 // trailing U+005D (]) removed.
436 input.remove_prefix(1);
437 input.remove_suffix(1);
438 return parse_ipv6(input);
439 }
440
441 // If isNotSpecial is true, then return the result of opaque-host parsing
442 // input.
443 if (!is_special()) {
444 return parse_opaque_host(input);
445 }
446 // Let domain be the result of running UTF-8 decode without BOM on the
447 // percent-decoding of input. Let asciiDomain be the result of running domain
448 // to ASCII with domain and false. The most common case is an ASCII input, in
449 // which case we do not need to call the expensive 'to_ascii' if a few
450 // conditions are met: no '%' and no 'xn-' subsequence.
451 std::string buffer = std::string(input);
452 // This next function checks that the result is ascii, but we are going to
453 // to check anyhow with is_forbidden.
454 // bool is_ascii =
455 unicode::to_lower_ascii(buffer.data(), buffer.size());
456 bool is_forbidden = unicode::contains_forbidden_domain_code_point(
457 buffer.data(), buffer.size());
458 if (is_forbidden == 0 && buffer.find("xn-") == std::string_view::npos) {
459 // fast path
460 host = std::move(buffer);
461 if (checkers::is_ipv4(host.value())) {
462 ada_log("parse_host fast path ipv4");
463 return parse_ipv4(host.value());
464 }
465 ada_log("parse_host fast path ", *host);
466 return true;
467 }
468 ada_log("parse_host calling to_ascii");
469 is_valid = ada::unicode::to_ascii(host, input, input.find('%'));
470 if (!is_valid) {
471 ada_log("parse_host to_ascii returns false");
472 return is_valid = false;
473 }
474
475 if (std::any_of(host.value().begin(), host.value().end(),
476 ada::unicode::is_forbidden_domain_code_point)) {
477 host = std::nullopt;
478 return is_valid = false;
479 }
480
481 // If asciiDomain ends in a number, then return the result of IPv4 parsing
482 // asciiDomain.
483 if (checkers::is_ipv4(host.value())) {
484 ada_log("parse_host got ipv4", *host);
485 return parse_ipv4(host.value());
486 }
487
488 return true;
489}
