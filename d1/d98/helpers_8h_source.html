203inline int leading_zeroes(uint32_t input_num) noexcept {
204#if ADA_REGULAR_VISUAL_STUDIO
205 unsigned long leading_zero(0);
206 unsigned long in(input_num);
207 return _BitScanReverse(&leading_zero, in) ? int(31 - leading_zero) : 32;
208#else
209 return __builtin_clz(input_num);
210#endif // ADA_REGULAR_VISUAL_STUDIO
211}

218inline int fast_digit_count(uint32_t x) noexcept {
219 auto int_log2 = [](uint32_t z) -> int {
220 return 31 - ada::helpers::leading_zeroes(z | 1);
221 };
222 // Compiles to very few instructions. Note that the
223 // table is static and thus effectively a constant.
224 // We leave it inside the function because it is meaningless
225 // outside of it (this comes at no performance cost).
226 const static uint64_t table[] = {
227 4294967296, 8589934582, 8589934582, 8589934582, 12884901788,
228 12884901788, 12884901788, 17179868184, 17179868184, 17179868184,
229 21474826480, 21474826480, 21474826480, 21474826480, 25769703776,
230 25769703776, 25769703776, 30063771072, 30063771072, 30063771072,
231 34349738368, 34349738368, 34349738368, 34349738368, 38554705664,
232 38554705664, 38554705664, 41949672960, 41949672960, 41949672960,
233 42949672960, 42949672960};
234 return int((x + table[int_log2(x)]) >> 32);
235}
